    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const onboardingContainer = document.getElementById('onboarding-container');
            const appContainer = document.getElementById('app-container');
            const formSteps = document.getElementById('form-steps');
            const stepContainers = document.querySelectorAll('.step-container');
            const progressBar = document.getElementById('progress-bar');
            const prevBtn = document.getElementById('prev-btn');
            const nextBtn = document.getElementById('next-btn');

            let currentStep = 0;
            const totalSteps = stepContainers.length;

            const incomeSlider = document.getElementById('income-slider');
            const incomeValue = document.getElementById('income-value');
            const savingsGoalSlider = document.getElementById('savings-goal-slider');
            const savingsGoalValue = document.getElementById('savings-goal-value');
            const spendingButtons = document.querySelectorAll('.spending-btn');

            const chatbotInput = document.getElementById('chatbot-input');
            const chatbotSendBtn = document.getElementById('chatbot-send-btn');
            const chatbotOutput = document.getElementById('chatbot-output');
           
            const fixedExpensesList = document.getElementById('fixed-expenses-list');
            const dailyExpensesList = document.getElementById('daily-expenses-list');
            const currentBalanceElement = document.getElementById('current-balance');
            const savingsGoalDisplay = document.getElementById('savings-goal-display');
           
            // New fixed expense inputs
            const addFixedExpenseBtn = document.getElementById('add-fixed-expense-btn');
            const fixedAmountInput = document.getElementById('fixed-amount-input');
            const fixedDescriptionInput = document.getElementById('fixed-description-input');
           
            // Daily expense inputs
            const addDailyExpenseBtn = document.getElementById('add-daily-expense-btn');
            const dailyAmountInput = document.getElementById('daily-amount-input');
            const dailyDescriptionInput = document.getElementById('daily-description-input');
           
            const savingsGoalAmountDisplay = document.getElementById('savings-goal-amount-display');
            const savedAmountDisplay = document.getElementById('saved-amount-display');
            const remainingAmountDisplay = document.getElementById('remaining-amount-display');
            const savingsProgressBar = document.getElementById('savings-progress-bar');
            const totalSavedKey = 'budgetwise_total_saved';
            const toastContainer = document.getElementById('toast-container');
           
            // New analysis elements
            const fixedSpendingPercent = document.getElementById('fixed-spending-percent');
            const dailySpendingPercent = document.getElementById('daily-spending-percent');
            const fixedExpensesBar = document.getElementById('fixed-expenses-bar');
            const dailyExpensesBar = document.getElementById('daily-expenses-bar');


            // --- Event Listeners for new interactive inputs ---
            if (incomeSlider) {
                incomeSlider.addEventListener('input', () => {
                    incomeValue.textContent = `€${incomeSlider.value}`;
                });
            }

            if (savingsGoalSlider) {
                savingsGoalSlider.addEventListener('input', () => {
                    savingsGoalValue.textContent = `${savingsGoalSlider.value}%`;
                });
            }

            spendingButtons.forEach(button => {
                button.addEventListener('click', () => {
                    spendingButtons.forEach(btn => btn.classList.remove('bg-indigo-600', 'text-white'));
                    button.classList.add('bg-indigo-600', 'text-white');
                    document.getElementById('average-spending').value = button.dataset.value;
                });
            });

            // --- FUNZIONI DI GESTIONE DELL'UI E DEI DATI ---
            const loadUserData = () => {
                const data = localStorage.getItem('budgetwise_user_data');
                return data ? JSON.parse(data) : null;
            };

            const saveUserData = (data) => {
                localStorage.setItem('budgetwise_user_data', JSON.stringify(data));
            };
           
            const loadFixedExpenses = () => {
                const data = localStorage.getItem('budgetwise_fixed_expenses');
                return data ? JSON.parse(data) : [];
            };

            const saveFixedExpenses = (expenses) => {
                localStorage.setItem('budgetwise_fixed_expenses', JSON.stringify(expenses));
            };

            const loadDailyExpenses = () => {
                const data = localStorage.getItem('budgetwise_daily_expenses');
                return data ? JSON.parse(data) : [];
            };

            const saveDailyExpenses = (expenses) => {
                localStorage.setItem('budgetwise_daily_expenses', JSON.stringify(expenses));
            };

            const loadTotalSaved = () => {
                const data = localStorage.getItem(totalSavedKey);
                return data ? parseFloat(data) : 0;
            };

            const saveTotalSaved = (amount) => {
                localStorage.setItem(totalSavedKey, amount.toFixed(2));
            };

            const showToast = (message) => {
                toastContainer.textContent = message;
                toastContainer.classList.add('show');
                setTimeout(() => {
                    toastContainer.classList.remove('show');
                }, 3000);
            };

            const showApp = (userData) => {
                onboardingContainer.style.display = 'none';
                appContainer.style.display = 'block';
                renderAppData(userData);
            };

            const renderAppData = (userData) => {
                const income = parseFloat(userData.income) || 0;
                const averageSpending = parseFloat(userData['average-spending']) || 0;
                const savingsGoal = parseInt(userData['savings-goal']) || 0;
                const savingsGoalAmount = parseFloat(userData['savings-goal-amount']) || 0;
                const fixedExpenses = loadFixedExpenses();
                const dailyExpenses = loadDailyExpenses();
                let totalSaved = loadTotalSaved();
           
                // Calcola il totale delle spese fisse
                let totalFixedExpenses = fixedExpenses.reduce((sum, expense) => sum + expense.amount, 0);

                // Calcola il totale delle spese giornaliere
                let totalDailyExpenses = dailyExpenses.reduce((sum, expense) => sum + expense.amount, 0);

                const totalExpenses = totalFixedExpenses + averageSpending + totalDailyExpenses;
                const balance = income - totalExpenses;

                currentBalanceElement.textContent = balance.toFixed(2);
                savingsGoalDisplay.textContent = savingsGoal;

                // Aggiorna la sezione dell'obiettivo di risparmio
                savingsGoalAmountDisplay.textContent = savingsGoalAmount.toFixed(2);
                let savedAmount = totalSaved;
                let remainingAmount = savingsGoalAmount - savedAmount;
                if (remainingAmount < 0) remainingAmount = 0;
               
                let progressPercentage = (savedAmount / savingsGoalAmount) * 100;
                if (progressPercentage > 100) progressPercentage = 100;

                savedAmountDisplay.textContent = savedAmount.toFixed(2);
                remainingAmountDisplay.textContent = remainingAmount.toFixed(2);
                savingsProgressBar.style.width = `${progressPercentage}%`;

                // Aggiorna l'analisi delle spese
                const totalVisibleExpenses = totalFixedExpenses + totalDailyExpenses;
                let fixedPercent = 0;
                let dailyPercent = 0;
               
                if (totalVisibleExpenses > 0) {
                    fixedPercent = (totalFixedExpenses / totalVisibleExpenses) * 100;
                    dailyPercent = (totalDailyExpenses / totalVisibleExpenses) * 100;
                }
               
                fixedSpendingPercent.textContent = `${fixedPercent.toFixed(0)}%`;
                dailySpendingPercent.textContent = `${dailyPercent.toFixed(0)}%`;
                fixedExpensesBar.style.width = `${fixedPercent}%`;
                dailyExpensesBar.style.width = `${dailyPercent}%`;
                dailyExpensesBar.style.left = `${fixedPercent}%`;

            const updateUI = () => {

                stepContainers[currentStep].classList.add('active');

                progressBar.style.width = `${progress}%`;
                prevBtn.style.display = currentStep === 0 ? 'none' : 'block';

            // --- LOGICA CHATBOT POTENZIATA CON LLM E RETRY PIÙ RESILIENTE ---
           
            async function fetchWithRetry(url, options, retries = 3) {
                for (let i = 0; i < retries; i++) {
                    try {
                        const response = await fetch(url, options);
                        if (response.ok) {
                            return await response.json();
                        } else {
                            throw new Error(`Errore HTTP: ${response.status}`);
                        }
                    } catch (error) {
                        const delay = Math.pow(2, i) * 1000;
                        console.error(`Tentativo ${i + 1} fallito. Riprovo in ${delay / 1000} secondi...`, error);
                        if (i < retries - 1) {
                            await new Promise(resolve => setTimeout(resolve, delay));
                        } else {
                            throw error;
                        }
                    }
                }
            }

            const handleSendMessage = async () => {
                const query = chatbotInput.value.trim();
                if (query === '') return;
               
                addMessageToChatbot(query, 'user');
                chatbotInput.value = '';

                const typingIndicator = document.createElement('p');
                typingIndicator.className = 'text-sm text-gray-500 mr-auto p-2';
                typingIndicator.textContent = 'Digitando...';
                chatbotOutput.appendChild(typingIndicator);
                chatbotOutput.scrollTop = chatbotOutput.scrollHeight;

                const userData = loadUserData();
                if (!userData) {
                    addMessageToChatbot("Errore: i dati utente non sono stati trovati. Completa l'onboarding per iniziare.", 'chatbot');
                    if (chatbotOutput.contains(typingIndicator)) {
                        chatbotOutput.removeChild(typingIndicator);
                    }
                    return;
                }

                const fixedExpenses = loadFixedExpenses();
                const totalFixedExpenses = fixedExpenses.reduce((sum, exp) => sum + exp.amount, 0);

                const systemPrompt = `Sei un consulente finanziario virtuale esperto e amichevole. Fornisci consigli finanziari personalizzati, rispondi alle domande sulla gestione del denaro e motiva l'utente a raggiungere i propri obiettivi di risparmio. Adotta un tono colloquiale, usa l'italiano e non presentarti come un modello di linguaggio. Hai un reddito di base di €${userData.income}, spese fisse per un totale di €${totalFixedExpenses}, una spesa media per cibo/svago di €${userData['average-spending']} e un obiettivo di risparmio mensile del ${userData['savings-goal']}%. Il tuo obiettivo di risparmio totale è di €${userData['savings-goal-amount']}. Utilizza questi dati per personalizzare le tue risposte.`;
                const userQuery = `L'utente chiede: ${query}. Fornisci una risposta utile e pertinente basata sui suoi dati finanziari e sul suo obiettivo.`;

                const apiKey = "";
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

                const payload = {
                    contents: [{ parts: [{ text: userQuery }] }],
                    tools: [{ "google_search": {} }],
                    systemInstruction: {
                        parts: [{ text: systemPrompt }]
                    },
                };

                try {
                    const result = await fetchWithRetry(apiUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    const candidate = result.candidates?.[0];
                    const responseText = candidate?.content?.parts?.[0]?.text;
                   
                    if (responseText) {
                        addMessageToChatbot(responseText, 'chatbot');
                    } else {
                        addMessageToChatbot("Mi dispiace, c'è stato un problema con la risposta. Potresti riformulare la domanda?", 'chatbot');
                    }
                } catch (error) {
                    console.error("Errore irrecuperabile durante la chiamata all'API:", error);
                    addMessageToChatbot("Ops, sembra che ci sia un problema di sicurezza del browser che sta bloccando la risposta. Prova a ricaricare la pagina o riprova tra poco. Mi scuso per l'inconveniente!", 'chatbot');
                } finally {
                    if (chatbotOutput.contains(typingIndicator)) {
                        chatbotOutput.removeChild(typingIndicator);
                    }
                }
            };

            const addMessageToChatbot = (message, sender = 'user') => {
                const messageDiv = document.createElement('div');
                messageDiv.className = `p-2 rounded-lg my-1 max-w-xs ${sender === 'user' ? 'bg-indigo-100 text-indigo-800 ml-auto' : 'bg-gray-200 text-gray-800 mr-auto'}`;
                messageDiv.textContent = message;
                chatbotOutput.appendChild(messageDiv);
                chatbotOutput.scrollTop = chatbotOutput.scrollHeight;
            };

            // Funzione per aggiornare i risparmi totali
            const updateMonthlySavings = () => {
                const lastUpdate = localStorage.getItem('last_update_date');
                const today = new Date();
                const currentMonth = today.getMonth() + 1;
                const currentYear = today.getFullYear();
               
                if (lastUpdate) {
                    const lastUpdateDate = new Date(lastUpdate);
                    const lastUpdateMonth = lastUpdateDate.getMonth() + 1;
                    const lastUpdateYear = lastUpdateDate.getFullYear();

                    // Se è un nuovo mese, calcola e aggiungi il surplus
                    if (currentMonth !== lastUpdateMonth || currentYear !== lastUpdateYear) {
                        const userData = loadUserData();
                        const fixedExpenses = loadFixedExpenses();
                        const dailyExpenses = loadDailyExpenses();
                        if (userData) {
                            const income = parseFloat(userData.income) || 0;
                            const averageSpending = parseFloat(userData['average-spending']) || 0;

                            const totalFixedExpenses = fixedExpenses.reduce((sum, exp) => sum + exp.amount, 0);
                            const totalDailyExpenses = dailyExpenses.reduce((sum, expense) => sum + expense.amount, 0);
                            const monthlySurplus = income - (totalFixedExpenses + averageSpending + totalDailyExpenses);

                            if (monthlySurplus > 0) {
                                let totalSaved = loadTotalSaved();
                                totalSaved += monthlySurplus;
                                saveTotalSaved(totalSaved);
                            }
                        }
                    }
                }
                localStorage.setItem('last_update_date', today.toISOString());
            };

                    }).catch(error => {

                    });
                });
            }

            // --- LOGICA ALL'AVVIO ---

            if (savedData) {
                updateMonthlySavings();
                showApp(savedData);
                renderDailyExpenses();
            } else {
                updateUI();
            }
        });
